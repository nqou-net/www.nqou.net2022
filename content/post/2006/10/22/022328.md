use strict;
use warnings;

use CGI qw(:standard);

my $styles = ["../../ipp.css"];

print header({ -charset => "utf-8" });
print start_html({
        -title => "テスト１",
        -style => { src => $styles },
    });
print p("うまく動くかな？n");

print div(
        a({ href => "../../../index.htm" }, "Home"),
        "/",
        a({ href => "../../" }, "Perl"),
        "/",

    );

        a({ href => q{mailto:nobu3@x.age.ne.jp}}, "Written by Nobu3"),
    );
