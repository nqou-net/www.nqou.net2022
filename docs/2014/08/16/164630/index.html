例えば最初からリファレンスで覚えるPerl

Aug 16, 2014 · 1 min read 
function show (args) {
    console.debug(args);
}

var str = 'string';
show(str);

// 配列
var array = [0, 1, 2]; // 配列の書き方で代入
show(array);

// オブジェクト
var obj = {key1: 'value1', key2: 'value2'}; // オブジェクトの書き方で代入
show(obj);

use DDP {deparse => 1};

sub show {
    my $args = shift;
    p $args;
}

my $str = 'string';
show($str);

my @array = (0, 1, 2);
show(\@array);

my %obj = (key1 => 'value1', key2 => 'value2');
show(\%obj);
