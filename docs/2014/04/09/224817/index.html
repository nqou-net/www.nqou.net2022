Mojoliciousでうっかりハマった件

Apr 9, 2014 · 1 min read 
use Mojolicious::Lite;
use Text::LTSV;

get '/' => sub {
  my $self = shift;
  my $p = Text::LTSV->new;
  my $datafile = 'hoge.ltsv';
  my $data = $p->parse_file($datafile) or die;
  $self->stash(data => $data);
  $self->render('index');
};

app->start;
__DATA__

@@ index.html.ep
% for my $line (@{$data}) {
    <a href="<%= $line->{url} %>"><%= $line->{url} %></a>
% }
