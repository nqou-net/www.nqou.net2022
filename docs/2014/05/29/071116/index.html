compress: {
    all: {
        options: {
            mode: 'gzip',
            level: 9
        },
        files: [{
            expand: true,
            cwd: '<%= config.dist %>',
            dest: '<%= config.dist %>',
            src: ['**/*.js'],
            ext: '.js.gz',
            extDot: 'last'
        },{
            expand: true,
            cwd: '<%= config.dist %>',
            dest: '<%= config.dist %>',
            src: ['**/*.css'],
            ext: '.css.gz',
            extDot: 'last'
        },{
            expand: true,
            cwd: '<%= config.dist %>',
            dest: '<%= config.dist %>',
            src: ['**/*.html'],
            ext: '.html.gz',
            extDot: 'last'
        }]
    }
},

grunt.registerTask('deploy', [
    'build',
    'compress:all',
    'rsync:prod'
]);
