Herokuでhypnotoadを使ってMojoliciousを起動する

May 9, 2015 · 1 min read 
{
    hypnotoad => {
        accepts            => 1000,
        backlog            => 10,
        clients            => 10,
        graceful_timeout   => 20,
        heartbeat_interval => 5,
        heartbeat_timeout  => 20,
        listen             => [qq{http://*:$ENV{PORT}}],
        proxy              => 1,
        requests           => 25,
        upgrade_timeout    => 60,
        workers            => 4,
    },
};

#!/usr/bin/env perl
use utf8;
use strict;
use warnings;
use Proclet;
use Furl;
require EV;
require IO::Socket::Socks;
require IO::Socket::SSL;


my $proclet = Proclet->new;

$proclet->service(
    code => ['hypnotoad', '-f', 'script/app'],
    tag  => 'web',
);

$proclet->service(
    code => ['perl', 'script/worker'],
    tag  => 'worker',
);

$proclet->service(
    code => sub {
        my $keepalive_url = $ENV{APP_KEEPALIVE_URL} or die 'set env APP_KEEPALIVE_URL.';
        Furl->new->get($keepalive_url);
    },
    tag   => 'keepalive',
    every => '*/30 * * * *',
);

$proclet->run;

web: carton exec -- perl app.pl
