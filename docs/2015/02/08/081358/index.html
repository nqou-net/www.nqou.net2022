Feb 8, 2015 · 1 min read

any '/' => sub {
    my ($c) = @_;
    my $config = $c->config;
    my $counter = $c->session->get('counter') || $config->{initial_counter};
    $counter++;
    $c->session->set('counter' => $counter);
    return $c->render('index.tx', {
        counter => $counter,
    });
};
