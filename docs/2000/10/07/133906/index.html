変換にかかる時間（その２）

Oct 7, 2000 · 1 min read 
                             <p>ソースコード</p>

#!/usr/bin/perl

use Benchmark;
use Jcode;

$count = 50000;

    $icode = 'sjis';
#    $icode = 'euc';
#    $icode = 'jis';
#    $icode = 'utf8';

$japanese = 'あいうえお阿衣兎得緒';
&amp;Jcode::convert($japanese, $icode);

print "Content-type: text/htmlnn";
print "<html><head><title></title></head>n";
print "<body>n";
print "<pre>n";

@t = timethese($count, {
     'to_SJIS' => '&amp;to_sjis;'
    ,'to_EUC'  => '&amp;to_euc;'
    ,'to_JIS'  => '&amp;to_jis;'
#    ,'to_UTF8' => '&amp;to_utf8;'
});

print "</pre>n";
print "</body></html>n";

exit(0);

sub to_sjis{
    &amp;Jcode::convert($japanese, 'sjis', $icode);
}

sub to_euc{
    &amp;Jcode::convert($japanese, 'euc',  $icode);
}

sub to_jis{
    &amp;Jcode::convert($japanese, 'jis',  $icode);
}

sub to_utf8{
    &amp;Jcode::convert($japanese, 'utf8', $icode);
}
